function show(id){document.querySelectorAll(".screen").forEach(s=>s.classList.remove("active"));document.getElementById(id).classList.add("active");}
function goHome(){person={};current=0;answers={};selectedIdx=null;confirmed=false;["f-name","f-title","f-dept","f-date","admin-pw"].forEach(id=>{const el=document.getElementById(id);if(el)el.value="";});show("s-register");}
function calcScore(ans){return Object.entries(ans).filter(([id,a])=>{const q=questions.find(q=>q.id==id);return q&&a===q.correct;}).length;}
function calcSections(ans){return sectionMeta.map(s=>{const sqs=questions.filter(q=>q.id>=s.range[0]&&q.id<=s.range[1]);const correct=sqs.filter(q=>ans[q.id]===q.correct).length;return{...s,correct,total:sqs.length,pct:Math.round(correct/sqs.length*100)};});}

// Apps Script entegrasyonu - GET parametreli
async function checkName(name){
  try{
    const url=APPS_SCRIPT_URL+"?action=check&name="+encodeURIComponent(name);
    const r=await fetch(url,{method:"GET",mode:"cors",redirect:"follow"});
    const d=await r.json();
    return d.exists||false;
  }catch(e){console.error("checkName error",e);return false;}
}
async function saveResult(data){
  try{
    const encoded=encodeURIComponent(JSON.stringify(data));
    const url=APPS_SCRIPT_URL+"?action=save&data="+encoded;
    await fetch(url,{method:"GET",mode:"cors",redirect:"follow"});
  }catch(e){console.error("saveResult error",e);}
}
async function loadResults(){
  try{
    const url=APPS_SCRIPT_URL+"?action=list";
    const r=await fetch(url,{method:"GET",mode:"cors",redirect:"follow"});
    const d=await r.json();
    return d.results||[];
  }catch(e){console.error("loadResults error",e);return[];}
}

async function startTest(){
  const name=document.getElementById("f-name").value;
  const title=document.getElementById("f-title").value;
  const dept=document.getElementById("f-dept").value;
  const date=document.getElementById("f-date").value;
  let valid=true;
  [["f-name","e-name","Ad Soyad zorunludur",name],["f-title","e-title","Unvan zorunludur",title],["f-dept","e-dept","Departman zorunludur",dept],["f-date","e-date","Tarih zorunludur",date]].forEach(([fi,ei,msg,val])=>{
    const inp=document.getElementById(fi);const err=document.getElementById(ei);
    if(!val.trim()){inp.classList.add("err");err.textContent="⚠ "+msg;err.style.display="block";valid=false;}
    else{inp.classList.remove("err");err.style.display="none";}
  });
  if(!valid)return;
  person={name,title,department:dept,date};
  show("s-loading");
  try{const exists=await checkName(norm(name));if(exists){document.getElementById("blocked-name").textContent=name;show("s-blocked");return;}}catch(e){console.error(e);}
  current=0;answers={};selectedIdx=null;confirmed=false;
  renderQuestion();show("s-test");
}

function renderQuestion(){
  const q=questions[current];
  const sec=sectionMeta.find(s=>q.id>=s.range[0]&&q.id<=s.range[1]);
  document.getElementById("t-section-label").textContent=q.icon+" "+q.section;
  document.getElementById("t-person-name").textContent=person.name;
  document.getElementById("t-counter").textContent=(current+1)+" / "+questions.length;
  document.getElementById("t-progress").style.width=(current/questions.length*100)+"%";
  const badge=document.getElementById("t-badge");
  badge.innerHTML=`<span style="font-size:14px">${q.icon}</span><span style="color:#e2e8f0;font-size:.74rem;font-weight:700;letter-spacing:.8px;text-transform:uppercase">${q.section}</span>`;
  badge.style.cssText=`background:${q.bg};border:1px solid ${q.color}88;display:inline-flex;align-items:center;gap:8px;border-radius:8px;padding:5px 14px;margin-bottom:20px`;
  document.getElementById("t-qnum").textContent="SORU "+q.id;
  document.getElementById("t-question").textContent=q.q;
  document.getElementById("t-options").innerHTML=q.opts.map((o,i)=>`<div class="option" id="opt-${i}" onclick="selectOption(${i})"><div class="opt-label" id="lbl-${i}">${String.fromCharCode(65+i)}</div><span class="opt-text">${o}</span></div>`).join("");
  document.getElementById("t-explanation").style.display="none";
  document.getElementById("btn-confirm").style.display="block";
  document.getElementById("btn-confirm").disabled=true;
  document.getElementById("btn-confirm").className="btn";
  document.getElementById("btn-next").style.display="none";
  selectedIdx=null;confirmed=false;
}

function selectOption(i){
  if(confirmed)return;
  selectedIdx=i;
  document.querySelectorAll(".option").forEach((el,j)=>{el.className="option"+(j===i?" selected":"");});
  document.getElementById("btn-confirm").disabled=false;
  document.getElementById("btn-confirm").className="btn btn-primary";
}

function confirmAnswer(){
  if(selectedIdx===null||confirmed)return;
  confirmed=true;
  const q=questions[current];
  answers[q.id]=selectedIdx;
  const ok=selectedIdx===q.correct;
  document.querySelectorAll(".option").forEach((el,j)=>{
    el.classList.add("locked");
    if(j===q.correct)el.classList.add("correct");
    else if(j===selectedIdx)el.classList.add("wrong");
  });
  const expDiv=document.getElementById("t-explanation");
  expDiv.className="explanation "+(ok?"ok":"nope");
  expDiv.innerHTML=`<strong style="color:${ok?"#4ade80":"#f87171"}">${ok?"✅ Doğru!":"❌ Yanlış"}</strong><br/><span style="color:#94a3b8;white-space:pre-line">${q.exp}</span>`;
  expDiv.style.display="block";
  document.getElementById("btn-confirm").style.display="none";
  document.getElementById("btn-next").style.display="block";
  document.getElementById("btn-next").textContent=current===questions.length-1?"Raporu Görüntüle 📋":"Sonraki Soru →";
}

async function nextQuestion(){
  if(current<questions.length-1){current++;renderQuestion();}
  else{await finishTest();}
}

async function finishTest(){
  show("s-saving");
  const score=calcScore(answers);
  const pct=Math.round(score/questions.length*100);
  const level=getLevel(pct);
  const ss=calcSections(answers);
  const data={name:person.name,title:person.title,department:person.department,date:person.date,score,total:questions.length,pct,level:level.label,sectionScores:ss.map(s=>({label:s.label,icon:s.icon,correct:s.correct,total:s.total,pct:s.pct})),answers:JSON.stringify(answers),submittedAt:new Date().toISOString()};
  try{await saveResult(data);}catch(e){console.error("Save error",e);}
  renderResult(data,ss,level,pct,score);
  show("s-result");
}

async function adminLogin(){
  const pw=document.getElementById("admin-pw").value;
  if(pw!==ADMIN_PASSWORD){document.getElementById("admin-err").textContent="⚠ Şifre yanlış.";document.getElementById("admin-err").style.display="block";return;}
  document.getElementById("admin-err").style.display="none";
  show("s-loading");
  try{const results=await loadResults();renderAdmin(results);show("s-admin");}catch(e){alert("Veriler yüklenemedi: "+e.message);show("s-admin-login");}
}
